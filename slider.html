up to the rescue

up pattern match

up function

                
up() {
    LIMIT=$1

    if [ -z "$LIMIT" ]; then
        LIMIT=1
    fi

    SEARCHPATH=$PWD

    # Non-numeric input, do pattern match
    if ! [[ "$LIMIT" =~ ^[0-9]+$ ]] ; then
        if ! [[ "$SEARCHPATH" =~ ^.*$LIMIT.*$ ]] ; then
            echo "expression not found in path"
        else
            while [ true ]; do
                SEARCHPATH=$SEARCHPATH/..
                cd $SEARCHPATH
                if [[ ${PWD##*/} =~ ^.*$LIMIT.*$ ]]; then
                    break;
                elif [[ -z ${PWD##*/} ]]; then
                    break;
                fi
            done
        fi
    else
        # go LIMIT dirs up
        for ((i=1; i <= LIMIT; i++))
            do
                SEARCHPATH=$SEARCHPATH/..
            done
        cd $SEARCHPATH
    fi
}
